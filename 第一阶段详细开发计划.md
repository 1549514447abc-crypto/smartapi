**1. 加密工具 (src/utils/encryption.ts)**

```typescript
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import crypto from 'crypto';
import config from '../config';

// 密码加密
export async function hashPassword(password: string): Promise<string> {
  const salt = await bcrypt.genSalt(10);
  return bcrypt.hash(password, salt);
}

// 密码验证
export async function comparePassword(
  password: string,
  hash: string
): Promise<boolean> {
  return bcrypt.compare(password, hash);
}

// 生成JWT Token
export function generateToken(payload: any, expiresIn?: string): string {
  return jwt.sign(payload, config.jwt.secret, {
    expiresIn: expiresIn || config.jwt.expiresIn
  });
}

// 验证JWT Token
export function verifyToken(token: string): any {
  try {
    return jwt.verify(token, config.jwt.secret);
  } catch (error) {
    throw new Error('Token无效或已过期');
  }
}

// 生成随机字符串
export function generateRandomString(length = 32): string {
  return crypto.randomBytes(length).toString('hex').substring(0, length);
}

// 生成推广码
export function generateReferralCode(length = 8): string {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
  let result = '';
  for (let i = 0; i < length; i++) {
    result += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return result;
}

// AES加密 (用于加密敏感数据)
export function encrypt(text: string): string {
  const algorithm = 'aes-256-cbc';
  const key = crypto.scryptSync(config.jwt.secret, 'salt', 32);
  const iv = crypto.randomBytes(16);

  const cipher = crypto.createCipheriv(algorithm, key, iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');

  return iv.toString('hex') + ':' + encrypted;
}

// AES解密
export function decrypt(text: string): string {
  const algorithm = 'aes-256-cbc';
  const key = crypto.scryptSync(config.jwt.secret, 'salt', 32);

  const parts = text.split(':');
  const iv = Buffer.from(parts[0], 'hex');
  const encrypted = parts[1];

  const decipher = crypto.createDecipheriv(algorithm, key, iv);
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');

  return decrypted;
}
```

**2. 响应工具 (src/utils/response.ts)**

```typescript
export interface ApiResponseData<T = any> {
  code: number;
  message: string;
  data?: T;
  timestamp?: number;
}

export class ApiResponse<T = any> {
  constructor(
    public code: number,
    public message: string,
    public data?: T
  ) {}

  static success<T>(data?: T, message = '操作成功'): ApiResponseData<T> {
    return {
      code: 200,
      message,
      data,
      timestamp: Date.now()
    };
  }

  static error(message = '操作失败', code = 500): ApiResponseData {
    return {
      code,
      message,
      timestamp: Date.now()
    };
  }

  static unauthorized(message = '未授权，请先登录'): ApiResponseData {
    return {
      code: 401,
      message,
      timestamp: Date.now()
    };
  }

  static forbidden(message = '没有权限访问'): ApiResponseData {
    return {
```

```typescript
// JWT认证中间件
export async function authenticateToken(
  req: Request,
  res: Response,
  next: NextFunction
) {
  try {
    // 从请求头获取token
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1]; // Bearer TOKEN

    if (!token) {
      return res.status(401).json(
        ApiResponse.unauthorized('缺少认证token')
      );
    }

    // 验证token
    const decoded = verifyToken(token);

    // 查询用户
    const user = await User.findByPk(decoded.userId);

    if (!user) {
      return res.status(401).json(
        ApiResponse.unauthorized('用户不存在')
      );
    }

    if (user.status !== 'active') {
      return res.status(403).json(
        ApiResponse.forbidden('账户已被禁用')
      );
    }

    // 将用户信息附加到请求对象
    req.user = user;
    next();
  } catch (error: any) {
    return res.status(401).json(
      ApiResponse.unauthorized(error.message || 'Token无效')
    );
  }
}

// 可选认证（token存在就验证，不存在也放行）
export async function optionalAuth(
  req: Request,
  res: Response,
  next: NextFunction
) {
  try {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1];

    if (token) {
      const decoded = verifyToken(token);
      const user = await User.findByPk(decoded.userId);
      if (user && user.status === 'active') {
        req.user = user;
      }
    }
    next();
  } catch (error) {
    // 忽略错误，继续执行
    next();
  }
```

```typescript
  async register(req: Request, res: Response) {
    try {
      const { username, email, phone, password } = req.body;

      // 1. 验证必填字段
      if (!username || !password) {
        return res.status(400).json(
          ApiResponse.badRequest('用户名和密码不能为空')
        );
      }

      if (!email && !phone) {
        return res.status(400).json(
          ApiResponse.badRequest('邮箱或手机号至少填写一个')
        );
      }

      // 2. 验证格式
      if (!isValidUsername(username)) {
        return res.status(400).json(
          ApiResponse.badRequest('用户名格式不正确（4-20位字母数字下划线）')
        );
      }

      if (!isValidPassword(password)) {
        return res.status(400).json(

        );

        return res.status(400).json(
          ApiResponse.badRequest('邮箱格式不正确')
        );
      }

      if (phone && !isValidPhone(phone)) {
        return res.status(400).json(
          ApiResponse.badRequest('手机号格式不正确')
        );
      }

      const existingUser = await User.findOne({ where: { username } });
      if (existingUser) {
        return res.status(400).json(
          ApiResponse.badRequest('用户名已存在')
        );
      }

      // 4. 检查邮箱是否已存在
      if (email) {
        const existingEmail = await User.findOne({ where: { email } });
        if (existingEmail) {
          return res.status(400).json(
            ApiResponse.badRequest('邮箱已被注册')
          );
        }
      }

      // 5. 检查手机号是否已存在
      if (phone) {
        const existingPhone = await User.findOne({ where: { phone } });
        if (existingPhone) {
          return res.status(400).json(
            ApiResponse.badRequest('手机号已被注册')
          );
        }
      }

      // 6. 创建用户
      const passwordHash = await hashPassword(password);
      const referralCode = generateReferralCode(8);
      const apiToken = `sk-${generateRandomString(32)}`;

      const user = await User.create({
        username,
        email,
        phone,
        passwordHash,
        referralCode,
        apiToken,
        credits: 10.00, // 注册送10积分
        nickname: username
      });

      await CreditTransaction.create({

        type: 'reward',
        amount: 10.00,
        balanceBefore: 0.00,
        balanceAfter: 10.00,
        description: '新用户注册奖励'

          user: user.toSafeObject()

        return res.status(400).json(
          ApiResponse.badRequest('账号和密码不能为空')
        );
      }

      // 1. 查找用户（支持用户名/邮箱/手机号登录）
      const user = await User.findOne({
        where: {
          [Op.or]: [
            { username: account },
            { email: account },
            { phone: account }
          ]
        }
      });

        return res.status(401).json(
          ApiResponse.unauthorized('账号或密码错误')
        );

      const isPasswordValid = await comparePassword(password, user.passwordHash);

      if (!isPasswordValid) {
        return res.status(401).json(
          ApiResponse.unauthorized('账号或密码错误')
        );
      }

      // 3. 检查用户状态
      if (user.status !== 'active') {
        return res.status(403).json(
          ApiResponse.forbidden('账户已被禁用')
        );
      }

      // 4. 更新登录信息
      await user.update({
        lastLoginAt: new Date(),
        lastLoginIp: req.ip
      });

      // 5. 生成token
      const token = generateToken({ userId: user.id });
```

**积分服务 (src/services/creditService.ts)**

```typescript
import { User } from '../models/User';
import { CreditTransaction } from '../models/CreditTransaction';
import { RechargeOrder } from '../models/RechargeOrder';
import { sequelize } from '../config/database';
import { Transaction, Op } from 'sequelize';

export class CreditService {
  /**
   * 扣除积分（带事务和锁）
   */
  async deductCredits(
    userId: number,
    amount: number,
    description: string,
    relatedType?: string,
    relatedId?: number
  ): Promise<{ success: boolean; message?: string }> {
    const t = await sequelize.transaction({
      isolationLevel: Transaction.ISOLATION_LEVELS.READ_COMMITTED
    });

    try {
      // 1. 使用行锁查询用户
      const user = await User.findByPk(userId, {
        lock: t.LOCK.UPDATE,
        transaction: t
      });

      if (!user) {
        await t.rollback();
        return { success: false, message: '用户不存在' };
      }

      // 2. 检查积分是否足够
      if (parseFloat(user.credits.toString()) < amount) {
        await t.rollback();
        return { success: false, message: '积分不足' };
      }

      const balanceBefore = parseFloat(user.credits.toString());
      const balanceAfter = balanceBefore - amount;

      // 3. 扣除积分
      await user.update(
        {
          credits: balanceAfter,
          totalConsumed: parseFloat(user.totalConsumed.toString()) + amount
        },
        { transaction: t }
      );

      // 4. 记录流水
      await CreditTransaction.create(
        {
          userId,
          type: 'consume',
          amount: -amount,
          balanceBefore,
          balanceAfter,
          relatedType,
          relatedId,
          description
        },
        { transaction: t }
      );

      await t.commit();
      return { success: true };
    } catch (error: any) {
      await t.rollback();
      throw error;
    }
  }

  /**
   * 增加积分
   */
  async addCredits(
    userId: number,
    amount: number,
    type: 'recharge' | 'reward' | 'refund' | 'commission' | 'adjustment',
    description: string,
    relatedType?: string,
    relatedId?: number
  ): Promise<void> {
    const t = await sequelize.transaction();

    try {
      const user = await User.findByPk(userId, {
        lock: t.LOCK.UPDATE,
        transaction: t
      });

      if (!user) {
        throw new Error('用户不存在');
      }

      const balanceBefore = parseFloat(user.credits.toString());
      const balanceAfter = balanceBefore + amount;

      // 更新积分
      await user.update(
        {
          credits: balanceAfter,
          totalRecharged: type === 'recharge'
            ? parseFloat(user.totalRecharged.toString()) + amount
            : user.totalRecharged
        },
        { transaction: t }
      );

      // 记录流水
      await CreditTransaction.create(
        {
          userId,
          type,
          amount,
          balanceBefore,
          balanceAfter,
          relatedType,
          relatedId,
          description
        },
        { transaction: t }
      );

      await t.commit();
    } catch (error) {
      await t.rollback();
      throw error;
    }
  }

  /**
   * 获取积分流水
   */
  async getTransactions(
    userId: number,
    page = 1,
    pageSize = 20,
    type?: string
  ) {
    const where: any = { userId };
    if (type) {
      where.type = type;
    }

    const { count, rows } = await CreditTransaction.findAndCountAll({
      where,
      order: [['createdAt', 'DESC']],
      limit: pageSize,
      offset: (page - 1) * pageSize
    });

    return {
      total: count,
      page,
      pageSize,
      totalPages: Math.ceil(count / pageSize),
      data: rows
    };
  }

  /**
   * 计算充值赠送积分
   */
  calculateBonus(amount: number): number {
    if (amount >= 1000) return amount * 0.2; // 充1000送200
    if (amount >= 500) return amount * 0.15;  // 充500送75
    if (amount >= 100) return amount * 0.1;   // 充100送10
    if (amount >= 50) return amount * 0.05;   // 充50送2.5
    return 0;
  }

  /**
   * 生成订单号
   */
  generateOrderNo(): string {
    const timestamp = Date.now();
    const random = Math.floor(Math.random() * 10000).toString().padStart(4, '0');
    return `RCH${timestamp}${random}`;
  }
}
```

**充值Controller (src/controllers/rechargeController.ts)**

```typescript
import { Request, Response } from 'express';
import { RechargeOrder } from '../models/RechargeOrder';
import { CreditService } from '../services/creditService';
import { AlipayService } from '../services/alipayService';
import { ApiResponse } from '../utils/response';
import config from '../config';

const creditService = new CreditService();
const alipayService = new AlipayService();

export class RechargeController {
  /**
   * 创建充值订单
   */
  async createOrder(req: Request, res: Response) {
    try {
      const { amount } = req.body;
      const userId = req.user.id;

      // 验证金额
      if (!amount || amount < 1) {
        return res.status(400).json(
          ApiResponse.badRequest('充值金额最少1元')
        );
      }

      if (amount > 10000) {
        return res.status(400).json(
          ApiResponse.badRequest('单笔充值最多10000元')
        );
      }

      // 计算赠送积分
      const credits = amount; // 1元 = 1积分
      const bonusCredits = creditService.calculateBonus(amount);
      const totalCredits = credits + bonusCredits;

      // 生成订单
      const orderNo = creditService.generateOrderNo();
      const order = await RechargeOrder.create({
        orderNo,
        userId,
        amount,
        credits: totalCredits,
        bonusCredits,
        paymentMethod: 'alipay',
        status: 'pending',
        expiredAt: new Date(Date.now() + 30 * 60 * 1000) // 30分钟过期
      });

      // 创建支付宝支付
      const payUrl = await alipayService.createPagePay({
        outTradeNo: orderNo,
        totalAmount: amount,
        subject: `积分充值 - ${totalCredits}积分`,
        body: `充值${amount}元，获得${totalCredits}积分（含赠送${bonusCredits}积分）`,
        returnUrl: `${config.frontendUrl}/recharge/success`,
        notifyUrl: `${config.backendUrl}/api/recharge/notify`
      });

      return res.json(
        ApiResponse.success({
          orderId: order.id,
          orderNo,
          amount,
          credits: totalCredits,
          bonusCredits,
          payUrl
        })
      );
    } catch (error: any) {
      console.error('创建充值订单失败:', error);
      return res.status(500).json(
        ApiResponse.error(error.message)
      );
    }
  }

  /**
   * 支付宝异步回调
   */
  async alipayNotify(req: Request, res: Response) {
    try {
      const postData = req.body;

      // 验证签名
      const signValid = alipayService.checkNotifySign(postData);
      if (!signValid) {
        return res.send('fail');
      }

      const {
        out_trade_no: orderNo,
        trade_no: transactionId,
        trade_status: tradeStatus,
        total_amount: totalAmount
      } = postData;

      // 查询订单
      const order = await RechargeOrder.findOne({ where: { orderNo } });
      if (!order) {
        return res.send('fail');
      }

      // 已处理过的订单，直接返回成功
      if (order.status === 'paid') {
        return res.send('success');
      }

      // 只处理支付成功的订单
      if (tradeStatus === 'TRADE_SUCCESS' || tradeStatus === 'TRADE_FINISHED') {
        // 验证金额
        if (parseFloat(totalAmount) !== parseFloat(order.amount.toString())) {
          console.error('订单金额不匹配:', orderNo, totalAmount, order.amount);
          return res.send('fail');
        }

        // 更新订单状态
        await order.update({
          status: 'paid',
          transactionId,
          paidAt: new Date()
        });

        // 增加积分
        await creditService.addCredits(
          order.userId,
          parseFloat(order.credits.toString()),
          'recharge',
          `充值订单: ${orderNo}`,
          'recharge_order',
          order.id
        );

        return res.send('success');
      }

      return res.send('fail');
    } catch (error: any) {
      console.error('支付宝回调处理失败:', error);
      return res.send('fail');
    }
  }

  /**
   * 查询订单状态
   */
  async queryOrder(req: Request, res: Response) {
    try {
      const { orderNo } = req.params;
      const userId = req.user.id;

      const order = await RechargeOrder.findOne({
        where: { orderNo, userId }
      });

      if (!order) {
        return res.status(404).json(
          ApiResponse.notFound('订单不存在')
        );
      }

      return res.json(
        ApiResponse.success(order)
      );
    } catch (error: any) {
      return res.status(500).json(
        ApiResponse.error(error.message)
      );
    }
  }

  /**
   * 获取充值记录
   */
  async getOrders(req: Request, res: Response) {
    try {
      const userId = req.user.id;
      const { page = 1, pageSize = 20, status } = req.query;

      const where: any = { userId };
      if (status) {
        where.status = status;
      }

      const { count, rows } = await RechargeOrder.findAndCountAll({
        where,
        order: [['createdAt', 'DESC']],
        limit: Number(pageSize),
        offset: (Number(page) - 1) * Number(pageSize)
      });

      return res.json(
        ApiResponse.success({
          total: count,
          page: Number(page),
          pageSize: Number(pageSize),
          totalPages: Math.ceil(count / Number(pageSize)),
          data: rows
        })
      );
    } catch (error: any) {
      return res.status(500).json(
        ApiResponse.error(error.message)
      );
    }
  }
}
```

```typescript
import axios from 'axios';
import { ElMessage } from 'element-plus';

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 10000
});

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器
request.interceptors.response.use(
  (response) => {
    const { code, message, data } = response.data;

    if (code === 200) {
      return data;
    } else {
      ElMessage.error(message || '请求失败');
      return Promise.reject(new Error(message));
    }
  },
  (error) => {
    if (error.response) {
      const { status, data } = error.response;

      if (status === 401) {
        ElMessage.error('登录已过期，请重新登录');
        localStorage.removeItem('token');
        window.location.href = '/login';
      } else {
        ElMessage.error(data.message || '网络错误');
      }
    } else {
      ElMessage.error('网络连接失败');
    }

    return Promise.reject(error);
  }
);
```
