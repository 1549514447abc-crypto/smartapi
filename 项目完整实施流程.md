# 创作魔方 Content Cube - 完整项目实施流程

## 📋 项目概览

### 项目目标
开发一个AI内容创作平台，第一期实现3个核心功能：
1. ✅ 视频文案提取
2. ✅ 工作流商店
3. ✅ 插件市场

### 技术栈选择
```
前端: Vue 3 + TypeScript + Vite + Element Plus
后端: Node.js + Express + TypeScript
数据库: MySQL 8.0 + Redis
ORM: Sequelize
支付: 微信支付 + 支付宝
存储: 阿里云OSS
部署: 阿里云ECS + Docker + Nginx
```

### 团队配置建议
- **后端开发**: 1人
- **前端开发**: 1人
- **产品/测试**: 0.5人（兼职或外包）

### 项目周期
**总计: 10-12周 (2.5-3个月)**

```typescript
import { Request, Response, NextFunction } from 'express';
import { verifyToken } from '../utils/encryption';
import { User } from '../models';
import { ApiResponse } from '../utils/response';

// 扩展Request类型
declare global {
  namespace Express {
    interface Request {
      user?: any;
    }
  }
}

export async function authenticateToken(
  req: Request,
  res: Response,
  next: NextFunction
) {
  try {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1]; // Bearer TOKEN

    if (!token) {
      return res.status(401).json(
        ApiResponse.unauthorized('缺少认证token')
      );
    }

    const decoded = verifyToken(token);
    const user = await User.findByPk(decoded.userId);

    if (!user || user.status !== 'active') {
      return res.status(401).json(
        ApiResponse.unauthorized('用户不存在或已被禁用')
      );
    }

    req.user = user;
    next();
  } catch (error) {
    return res.status(401).json(
      ApiResponse.unauthorized('Token无效或已过期')
    );
  }
}

// 检查管理员权限
export function requireAdmin(

  }
  next();
}
```

```typescript
import { Request, Response, NextFunction } from 'express';
import { ApiResponse } from '../utils/response';

export function errorHandler(
  err: any,
  req: Request,
  res: Response,
  next: NextFunction
) {
  console.error('错误详情:', err);

  // Sequelize错误
  if (err.name === 'SequelizeValidationError') {
    return res.status(400).json(
      ApiResponse.badRequest(err.errors[0].message)
    );
  }

  if (err.name === 'SequelizeUniqueConstraintError') {
    return res.status(400).json(
      ApiResponse.badRequest('数据已存在')
    );
  }

  // JWT错误
  if (err.name === 'JsonWebTokenError') {
    return res.status(401).json(
      ApiResponse.unauthorized('Token无效')
    );
  }

  // 默认错误
  return res.status(err.status || 500).json(
    ApiResponse.error(err.message || '服务器内部错误', err.status || 500)
  );
}
```

### Week 5: 积分和财务系统

#### 积分充值

**src/services/creditService.ts:**
```typescript
import { User, CreditTransaction, RechargeOrder } from '../models';
import { sequelize } from '../config/database';

export class CreditService {
  // 充值积分
  async recharge(userId: number, amount: number, paymentMethod: string) {
    const transaction = await sequelize.transaction();

    try {
      const user = await User.findByPk(userId, { transaction });

      if (!user) {
        throw new Error('用户不存在');
      }

      // 创建充值订单
      const orderNo = this.generateOrderNo();
      const credits = amount; // 1元 = 1积分
      const bonusCredits = this.calculateBonus(amount); // 计算赠送积分

      const order = await RechargeOrder.create({
        orderNo,
        userId,
        amount,
        credits,
        bonusCredits,
        paymentMethod,
        status: 'pending'
      }, { transaction });

      await transaction.commit();

      return order;
    } catch (error) {
      await transaction.rollback();
      throw error;
    }
  }

  // 支付成功回调
  async handlePaymentSuccess(orderNo: string) {
    const transaction = await sequelize.transaction();

    try {
      const order = await RechargeOrder.findOne({
        where: { orderNo }
      });

      if (!order) {
        throw new Error('订单不存在');
      }

      if (order.status !== 'pending') {
        throw new Error('订单状态异常');
      }

      const user = await User.findByPk(order.userId, { transaction });
      if (!user) {
        throw new Error('用户不存在');
      }

      const totalCredits = parseFloat(order.credits as any) + parseFloat(order.bonusCredits as any);
      const balanceBefore = parseFloat(user.credits as any);
      const balanceAfter = balanceBefore + totalCredits;

      // 更新用户积分
      await user.update({
        credits: balanceAfter,
        totalRecharged: parseFloat(user.totalRecharged as any) + parseFloat(order.amount as any)
      }, { transaction });

      // 记录积分流水
      await CreditTransaction.create({
        userId: order.userId,
        type: 'recharge',
        amount: totalCredits,
        balanceBefore,
        balanceAfter,
        relatedType: 'recharge_order',
        relatedId: order.id,
        description: `充值 ${order.amount}元，获得${totalCredits}积分`
      }, { transaction });

      // 更新订单状态
      await order.update({
        status: 'paid',
        paidAt: new Date()
      }, { transaction });

      await transaction.commit();

      return { user, order };
    } catch (error) {
      await transaction.rollback();
      throw error;
    }
  }

  // 扣除积分
  async deductCredits(
    userId: number,
    amount: number,
    relatedType: string,
    relatedId: number,
    description: string
  ) {
    const transaction = await sequelize.transaction();

    try {
      const user = await User.findByPk(userId, {
        transaction,
        lock: transaction.LOCK.UPDATE
      });

      if (!user) {
        throw new Error('用户不存在');
      }

      const currentCredits = parseFloat(user.credits as any);

      if (currentCredits < amount) {
        throw new Error('积分不足');
      }

      const balanceAfter = currentCredits - amount;

      // 更新用户积分
      await user.update({
        credits: balanceAfter,
        totalConsumed: parseFloat(user.totalConsumed as any) + amount
      }, { transaction });

      // 记录流水
      await CreditTransaction.create({
        userId,
        type: 'consume',
        amount: -amount,
        balanceBefore: currentCredits,
        balanceAfter,
        relatedType,
        relatedId,
        description
      }, { transaction });

      await transaction.commit();

      return user;
    } catch (error) {
      await transaction.rollback();
      throw error;
    }
  }

  // 生成订单号
  private generateOrderNo(): string {
    const now = new Date();
    const year = now.getFullYear();
    const month = String(now.getMonth() + 1).padStart(2, '0');
    const day = String(now.getDate()).padStart(2, '0');
    const hours = String(now.getHours()).padStart(2, '0');
    const minutes = String(now.getMinutes()).padStart(2, '0');
    const seconds = String(now.getSeconds()).padStart(2, '0');
    const random = Math.floor(Math.random() * 10000).toString().padStart(4, '0');

    return `R${year}${month}${day}${hours}${minutes}${seconds}${random}`;
  }

  // 计算赠送积分
  private calculateBonus(amount: number): number {
    if (amount >= 1000) return amount * 0.2; // 充值1000+赠送20%
    if (amount >= 500) return amount * 0.15;  // 充值500+赠送15%
    if (amount >= 100) return amount * 0.1;   // 充值100+赠送10%
    return 0;
  }
}
```

```bash
# 创建项目目录
mkdir -p /var/www/smartapi-backend
cd /var/www/smartapi-backend

# 克隆代码 (或使用FTP上传)
git clone your-repo-url .

# 安装依赖
npm install --production

# 配置环境变量
cp .env.example .env
vim .env  # 修改为生产环境配置

# 运行数据库迁移
npm run migrate

# 使用PM2启动

pm2 save
pm2 startup
```

#### 4. 配置Nginx

```nginx
# /etc/nginx/conf.d/smartapi.conf

server {
    listen 80;
    server_name api.yourdomain.com;

    location / {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_cache_bypass $http_upgrade;
    }
}

server {
    listen 80;
    server_name www.yourdomain.com;
    root /var/www/smartapi-frontend/dist;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location /api {
        proxy_pass http://localhost:5000;
    }
}
```

```bash
# 重启Nginx
nginx -t
systemctl reload nginx
```

#### 5. 配置SSL证书 (Let's Encrypt)

```bash
# 安装certbot
yum install certbot python3-certbot-nginx -y

# 申请证书
certbot --nginx -d yourdomain.com -d www.yourdomain.com -d api.yourdomain.com

# 自动续期
echo "0 0 1 * * /usr/bin/certbot renew --quiet" | crontab -
```
