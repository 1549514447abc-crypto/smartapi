ALIPAY_APP_ID=2021006115621581
ALIPAY_GATEWAY=https://openapi.alipay.com/gateway.do

# 证书文件路径（或直接存证书内容）
ALIPAY_APP_PRIVATE_KEY=应用私钥内容
ALIPAY_APP_CERT_PATH=证书存放路径/appCertPublicKey_2021006115621581.crt
ALIPAY_ALIPAY_CERT_PATH=证书存放路径/alipayCertPublicKey_RSA2.crt
ALIPAY_ROOT_CERT_PATH=证书存放路径/alipayRootCert.crt

# 回调地址
ALIPAY_NOTIFY_URL=异步通知地址
ALIPAY_RETURN_URL=同步跳转地址项目交付时请提供：

配置文件的具体位置（比如 .env 文件路径）
