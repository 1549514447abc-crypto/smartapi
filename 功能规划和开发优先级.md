```typescript
async function deductCredits(userId, amount, description) {
  const transaction = await sequelize.transaction();
  try {
    // 1. 锁定用户记录
    const user = await User.findByPk(userId, {
      transaction,
      lock: transaction.LOCK.UPDATE
    });

    // 2. 检查余额
    if (user.credits < amount) {
      throw new Error('积分不足');
    }

    // 3. 扣除积分
    await user.update({
      credits: user.credits - amount
    }, { transaction });

    // 4. 记录流水
    await CreditTransaction.create({
      userId,
      amount: -amount,
      description
    }, { transaction });

    await transaction.commit();
  } catch (error) {
    await transaction.rollback();
    throw error;
  }
}
```

```typescript
// 视频提取服务
export class VideoExtractService {
  // 创建提取任务
  async createTask(userId: number, url: string, options: any) {
    // 1. 识别平台
    const platform = this.detectPlatform(url);

    // 2. 检查用户积分
    const estimatedCredits = await this.estimateCredits(url);
    await this.checkUserCredits(userId, estimatedCredits);

    // 3. 创建任务
    const task = await VideoExtractTask.create({
      taskNo: this.generateTaskNo(),
      userId,
      videoUrl: url,
      platform,
      enableCorrection: options.enableCorrection,
      status: 'pending'
    });

    // 4. 放入队列
    await videoQueue.add({
      taskId: task.id,
      userId,
      url,
      options
    });

    return task;
  }

  // 处理视频提取（Worker执行）
  async processTask(taskId: number) {
    const task = await VideoExtractTask.findByPk(taskId);

    try {
      await task.update({ status: 'processing', startedAt: new Date() });

      // 1. 解析视频信息
      const videoInfo = await this.parseVideo(task.videoUrl);

      await task.update({
        videoTitle: videoInfo.title,
        videoDuration: videoInfo.duration
      });

      // 2. 提取字幕/转文字
      let content = '';
      if (videoInfo.hasSubtitle) {
        // 直接提取字幕
        content = await this.extractSubtitle(videoInfo);
      } else {
        // ASR语音识别
        content = await this.speechToText(videoInfo.audioUrl);
      }

      // 3. AI纠错（如果启用）
      if (task.enableCorrection) {
        content = await this.aiCorrection(content);
      }

      // 4. 生成Word文档
      const wordUrl = await this.generateWordDoc(content, task.id);

      // 5. 计算消耗积分
      const creditsUsed = this.calculateCredits(
        videoInfo.duration,
        task.enableCorrection
      );

      // 6. 扣除积分
      await creditService.deductCredits(
        task.userId,
        creditsUsed,
        'video_extract',
        task.id,
        `视频文案提取: ${task.videoTitle}`
      );

      // 7. 更新任务状态
      await task.update({
        status: 'completed',
        content,
        wordCount: content.length,
        wordFileUrl: wordUrl,
        creditsUsed,
        completedAt: new Date()
      });

      // 8. 发送通知
      await this.notifyUser(task.userId, task.id);

    } catch (error) {
      await task.update({
        status: 'failed',
        errorMessage: error.message
      });
      throw error;
    }
  }

  // 调用ASR API
  async speechToText(audioUrl: string): Promise<string> {
    // 使用阿里云ASR示例
    const client = new ALiYunASR({
      accessKeyId: process.env.ALI_ACCESS_KEY_ID,
      accessKeySecret: process.env.ALI_ACCESS_KEY_SECRET
    });

    const result = await client.recognize(audioUrl);
    return result.text;
  }
}
```

#### 插件调用统一接口

```typescript
// 插件执行服务
export class PluginExecutionService {
  async execute(pluginSlug: string, userId: number, parameters: any) {
    // 1. 获取插件配置
    const plugin = await Plugin.findOne({ where: { slug: pluginSlug } });

    // 2. 获取用户API密钥
    const userConfig = await UserPluginConfig.findOne({
      where: { userId, pluginId: plugin.id }
    });

    if (!userConfig) {
      throw new Error('请先配置API密钥');
    }

    // 3. 解密API密钥
    const apiKeys = JSON.parse(decrypt(userConfig.apiKeysEncrypted));

    // 4. 验证参数
    this.validateParameters(plugin.parameters, parameters);

    // 5. 计算费用
    const credits = this.calculateCredits(plugin.pricing, parameters);

    // 6. 检查积分
    await creditService.checkCredits(userId, credits);

    // 7. 调用第三方API
    const startTime = Date.now();
    let result;

    try {
      result = await this.callPluginApi(plugin, apiKeys, parameters);
    } catch (error) {
      // 记录失败日志
      await PluginUsageLog.create({
        userId,
        pluginId: plugin.id,
        status: 'failed',
        errorMessage: error.message
      });
      throw error;
    }

    const executionTime = Date.now() - startTime;

    // 8. 扣除积分
    await creditService.deductCredits(
      userId,
      credits,
      'plugin_usage',
      plugin.id,
      `使用插件: ${plugin.name}`
    );

    // 9. 记录使用日志
    await PluginUsageLog.create({
      userId,
      pluginId: plugin.id,
      requestParams: parameters,
      responseData: result,
      creditsUsed: credits,
      executionTime,
      status: 'success'
    });

    // 10. 更新统计
    await plugin.increment('usageCount');

    return result;
  }

  // 调用不同provider的API
  async callPluginApi(plugin, apiKeys, parameters) {
    switch (plugin.provider) {
      case 'replicate':
        return await this.callReplicateApi(plugin, apiKeys, parameters);
      case 'openai':
        return await this.callOpenAiApi(plugin, apiKeys, parameters);
      // ... 其他provider
    }
  }
}
```

```typescript
// 发送消息API
router.post('/api/ai/conversations/:id/messages', async (req, res) => {
  const conversationId = req.params.id;
  const { content } = req.body;
  const userId = req.user.id;

  // 1. 获取会话和智能体
  const conversation = await AiConversation.findByPk(conversationId);
  const agent = await AiAgent.findByPk(conversation.agentId);

  // 2. 计算轮次和费用
  const roundNumber = Math.floor(conversation.messageCount / 2) + 1;
  const credits = calculateRoundCredits(agent.pricingConfig, roundNumber);

  // 3. 检查积分
  await creditService.checkCredits(userId, credits);

  // 4. 保存用户消息
  const userMessage = await AiMessage.create({
    conversationId,
    userId,
    agentId: agent.id,
    role: 'user',
    content,
    roundNumber
  });

  // 5. 获取上下文
  const context = await this.getContext(conversationId, agent.features.maxContextRounds);

  // 6. 调用AI API
  const aiResponse = await aiService.chat({
    model: agent.model,
    systemPrompt: agent.systemPrompt,
    messages: [
      ...context,
      { role: 'user', content }
    ]
  });

  // 7. 保存AI回复
  const assistantMessage = await AiMessage.create({
    conversationId,
    userId,
    agentId: agent.id,
    role: 'assistant',
    content: aiResponse.content,
    roundNumber,
    promptTokens: aiResponse.usage.promptTokens,
    completionTokens: aiResponse.usage.completionTokens,
    totalTokens: aiResponse.usage.totalTokens,
    creditsUsed: credits
  });

  // 8. 扣除积分
  await creditService.deductCredits(
    userId,
    credits,
    'ai_message',
    assistantMessage.id,
    `智能体对话: ${agent.name} - 第${roundNumber}轮`
  );

  // 9. 更新会话统计
  await conversation.increment('messageCount', { by: 2 });
  await conversation.increment('totalCredits', { by: credits });

  // 10. 返回结果
  res.json({
    message: assistantMessage,
    usage: {
      roundNumber,
      creditsUsed: credits,
      nextRoundPrice: calculateRoundCredits(agent.pricingConfig, roundNumber + 1)
    }
  });
});
```

## 💡 MVP最小化建议

### 方案A: 极简MVP (4-6周)

**目标**: 最快速度验证核心需求

**只做**:
```
Week 1-2: 基础系统
├── 用户注册登录
├── 积分充值（仅支付宝）
└── 简单的用户中心

Week 3-4: 核心功能
├── 视频文案提取（仅抖音+B站）
│   ├── 不做批量
│   ├── 不做AI纠错
│   └── 仅导出TXT
└── 简单管理后台

Week 5-6: 测试上线
├── 功能测试
├── 部署上线
└── 小范围内测
```

**暂不做**:
- ❌ 工作流商店
- ❌ 插件市场
- ❌ 智能体系统
- ❌ 会员系统
- ❌ 推广分销
